Criando o modelo

library(randomForest)
## randomForest 4.6-10
## Type rfNews() to see new features/changes/bug fixes.
formula <- class ~ x1 + y1 + z1 + x2 + y2 + z2 + x3 + y3 + z3 + x4 + y4 + z4
model <- randomForest(formula, data=treinamento, do.trace=100, importance=TRUE)
